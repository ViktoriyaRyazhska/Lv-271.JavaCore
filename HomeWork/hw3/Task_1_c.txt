package Homework;

import java.util.Scanner;

public class Task_1_c {
	static Scanner scanner = new Scanner(System.in);
	public static void main(String[] args) {
		System.out.println("enter number of HTTP Error");
		numder=scanner.nextInt();
errorName();
	}

	public static enum HTTPError {
		Bad_Request,Unauthorized,Payment_Required,Forbidden 
		
	}
	static int numder;
	
	public static void errorName() { 
		HTTPError hError = null ;
		
		switch(numder){
		case 400: hError=HTTPError.Bad_Request;  break;
		case 401: hError=HTTPError.Unauthorized;break;
		case 402: hError=HTTPError.Payment_Required;break;
		case 403: hError=HTTPError.Forbidden;break;
		default :System.out.println("incorrect number or not have information ");
		return;
		}
		System.out.println("name of error "+hError);
	}
	
	
}
