package homeWork;


public class Dog {

	private String name;
	private double age;
	private breedOfDogs breed;
	public enum breedOfDogs {AKITA,ALASKAN_MALAMUTE,AMERICAN_FOXHOUND,AIREDALE_TERRIER};

	public  Dog(){}
	public  Dog(String name ,double age,breedOfDogs breed){
		this.name=name;
		this.age=age;
		this.breed=breed;
		
		
	}
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	
	
	public boolean equals(Dog obj) {
		
		if(super.equals(obj)){return true;}
		else if(this.name==obj.name){return true;}
		else return false;
		
	
	}
	public void maxAge(Dog dog2,Dog dog3){
		Dog max=new Dog();
		if(this.age>dog2.age){max=this;}else{max=dog2;}
		if(max.age<dog3.age){max=dog3;}
		System.out.println("max age hawe \n"+max.name+"\n"+max.breed);
	}
}
